一键创建php开发环境

开发环境版本

nginx 1.16.0

php 7.3.6

mysql 5.6.17

redis 3.2.3

rabbitmq 3.6.1

mongo 3.4.10

架构

vue前端所用nginx

php后端所用的phpcli、php-fpm+nginx

redis

rabbitmq

mysql

mongodb

docker-commpose.yml

version: '3'
services:

  nginx:
    networks:
      - newdev
    image: registry-intl.cn-shanghai.aliyuncs.com/mywallet/dev-nginx:1.16.0

# 80:80 指将容器nginx的80端口映射到宿主机的80端口。宿主机端口请根据自己的开发环境修改
    ports:
      - "80:80"
    volumes:

# vue前端nginx的配置文件挂载；./docker/nginx/default.conf指当前目录下的docker/nginx/default.conf,可自定义
      - "./docker/nginx/default.conf:/etc/nginx/conf.d/default.conf"

# vue前端代码挂载,可自定义
      - "/home/wwwroot/www/acf_vue/vue/dist:/usr/share/nginx/html"

  phpcli:
    networks:
      - newdev
    image: registry-intl.cn-shanghai.aliyuncs.com/mywallet/dev-phpcli:7.3.6
    ports:
      - "9999:9999"
    volumes:
      - "/home/wwwroot/www/acf_hhh:/data/www"

  cnp:
    networks:
      - newdev
    image: registry-intl.cn-shanghai.aliyuncs.com/mywallet/dev-cnp:latest
    ports:
      - "9000:9000"
    volumes:

# php后端nginx的配置文件挂载；./docker/cnp/conf/nginx.conf指当前目录下的docker/cnp/conf/nginx.conf,可自定义
      - "./docker/cnp/conf/nginx.conf:/usr/local/nginx/conf/nginx.conf"

# php后端代码挂载,可自定义
      - "/home/wwwroot/www/acf_hhh:/home/wwwroot/www/"

  mysql:
    networks:
      - newdev
    image: mysql:5.6.17

# 3306:3306 指将mysql容器的3306端口映射到宿主机的3306端口，宿主机端口请根据自己的开发环境修改
# 外部可通过客户端访问mysql服务 宿主机ip + 自定义port；
# 代码内部访问mysql服务 容器名（mysql） + 3306
# root默认密码为12346，可在容器启动之前自定义
# mysql数据在不删除本地挂载的情况，重启容器不会被删除

    ports:
      - "3306:3306"
    volumes:
      - "/data/db_mysql/data:/var/lib/mysql"
    environment:
      - MYSQL_ROOT_PASSWORD=123456

  redis:
    networks:
      - newdev
    image: redis:3.2.3
    command: redis-server --requirepass "123456" --appendonly yes

# 6379:6379 指将redis容器的6379端口映射到宿主机的6379端口，宿主机端口请根据自己的开发环境修改
# 外部可通过客户端访问redis服务 宿主机ip + 自定义port；
# 代码内部访问redis服务 容器名（redis） + 6379
# 默认密码为12346，可在容器启动之前自定义
# redis数据在不删除本地挂载的情况，重启容器不会被删除

    ports:
      - "6379:6379"
    volumes:
      - /data/db_redis:/data

  mongodb:
    networks:
      - newdev
    image: mongo:3.4.10
    command: [--auth]

# 27017:27017 指将mongodb容器的27017端口映射到宿主机的27017端口，宿主机端口请根据自己的开发环境修改
# 外部可通过客户端访问redis服务 宿主机ip + 自定义port；
# 代码内部访问mongodb服务 容器名（mongodb） + 27017
# 相关库以及账号密码配置，请查看 注意事项
# redis数据在不删除本地挂载的情况，重启容器不会被删除

    ports:
      - "27017:27017"
    volumes:
      - /data/db_mongo:/data/db

  rabbitmq:
    networks:
      - newdev
    image: rabbitmq:3.7.0-management

# 5672:5672 指将mongodb容器的5672端口映射到宿主机的5672端口，宿主机端口请根据自己的开发环境修改
# 外部可通过客户端访问rabbitmq服务 宿主机ip + 自定义port；
# 代码内部访问mongodb服务 容器名（rabbitmq） + 5672
# 15672是rabbitmq web的端口，可通过宿主机ip + 自定义访问，账号密码均为 guest

    ports:
      - "5672:5672"
      - "15672:15672"
    volumes:
      - "/data/db_rabbitmq:/var/lib/rabbitmq"

networks:
  newdev:

docker-compose安装：

curl -L https://github.com/docker/compose/releases/download/1.24.0/docker-compose-`uname -s`-`uname -m` -o /usr/local/bin/docker-compose && chmod u+x /usr/local/bin/docker-compose

启动容器：

docker-compose up -d

删除容器：

docker-compose down

注意事项：

  • mysql启动之后，需要通过客户端连接到mysql，新建项目对应的database

  • mongodb启动之后，可通过portainer进入容器，新建项目对应的库以及账号密码

进入容器，输入mongo，回车

进入管理库 

use admin

在管理库中创建超级用户

db.createUser(
  {
    user: "toor",
    pwd: "123456",
    roles: [ { role: "root", db: "admin" } ]
  }
)

登录超级用户

db.auth('toor','123456')

进入新建库newacf

use newacf

在test库中创建用户

db.createUser(
  {
    user: "newacf",
    pwd: "123456",
    roles: [ { role: "dbOwner", db: "newacf" } ]
  }
)

登录test库的用户

db.auth('newacf','123456')

在test库任意新建一张表

db.newacf.insert({'x':1})

退出mongo客户端

exit
